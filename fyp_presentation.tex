\begin{frame}
	\frametitle{Why Underwater Sonar SLAM Is Hard}
	\framesubtitle{Can a renderable sonar map provide online correction factors?}

	\begin{columns}[T]
		\begin{column}{0.47\textwidth}
			\storycard{ICLBlue!6}{ICLBlue}{1. Problem setting}
			{Underwater robots cannot use GNSS. IMU/DVL odometry therefore drifts over time.}

			\vspace{1.8mm}
			\storycard{ICLBlue!6}{ICLBlue}{2. Sensor limitation}
			{Sonar works underwater, but the image evidence is sparse, ambiguous, and viewpoint-dependent.}

			\vspace{1.8mm}
			\storycard{ICLBlue!6}{ICLBlue}{3. Sonar-based SLAM limitation}
			{Geometric matching becomes fragile under sonar noise and repeated acoustic structure.}
		\end{column}
		\begin{column}{0.47\textwidth}
			\storycard{ICLCream}{ICLBlue}{4. Key insight from 3DGS}
			{A learned Gaussian scene can predict the sonar view from a candidate pose.}

			\vspace{1.8mm}
			\storycard{ICLCream}{ICLBlue}{5. Gap}
			{3DGS is rarely used as an online sonar measurement model inside SLAM.}

			\vspace{1.8mm}
			\storycard{ICLCream}{ICLBlue}{6. Project idea}
			{Use render consistency between live sonar and 3D Gaussian Splatting prediction as a SLAM constraint.}
		\end{column}
	\end{columns}

\end{frame}

\begin{frame}
	\frametitle{Research Gap: Renderable Sonar Maps Inside SLAM}
	\framesubtitle{What the literature gives, and what this project adds}

	\vspace{-1mm}
	\begin{columns}[T]
		\begin{column}{0.31\textwidth}
			{\small\textbf{Sonar SLAM}}
			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{0.8mm}
				\scriptsize
				\item Uses sonar features, occupancy maps, or image factors to reduce drift.
				\item Still depends on difficult acoustic data association.
				\item Examples: Westman, Li, McConnell, Aqua-SLAM.
			\end{itemize}
		\end{column}
		\begin{column}{0.31\textwidth}
			{\small\textbf{3DGS SLAM}}
			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{0.8mm}
				\scriptsize
				\item Uses Gaussian maps for dense tracking and reconstruction.
				\item Mostly assumes camera or RGB-D observations.
				\item Examples: Gaussian-SLAM, GS-SLAM, RTG-SLAM, SemGauss-SLAM.
			\end{itemize}
		\end{column}
		\begin{column}{0.31\textwidth}
			{\small\textbf{Underwater / sonar GS}}
			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{0.8mm}
				\scriptsize
				\item Learns renderable underwater or sonar-like scene models.
				\item Mainly evaluated as reconstruction or rendering.
				\item Examples: NAS-GS, Aqua-Splat, SeaSplat, WaterSplatting, UW-GS.
			\end{itemize}
		\end{column}
	\end{columns}

	\vspace{3mm}
	\begin{center}
		\setlength{\fboxsep}{4pt}
		\fcolorbox{ICLBlue}{ICLCream}{%
			\begin{minipage}{0.90\textwidth}
				\centering
				{\small\textbf{Missing bridge}}\par\vspace{0.5mm}
				{\scriptsize
				Use a trained sonar Gaussian map as an online measurement model:
				candidate Pose3 \(\rightarrow\) NAS-GS render \(\rightarrow\) live-sonar consistency check \(\rightarrow\) gated Pose3 factor in iSAM2.}
			\end{minipage}}
	\end{center}
\end{frame}

\begin{frame}
	\frametitle{NAS-GS: Noise-Aware Sonar Gaussian Splatting}
	\framesubtitle{A renderable sonar measurement model reference}

	\begin{columns}[T]
		\begin{column}{0.58\textwidth}
			\vspace{-1mm}
			\begin{center}
				\includegraphics[width=\linewidth]{nasgs_pipeline_overview.png}

				\vspace{0.5mm}
				{\scriptsize NAS-GS training pipeline, adapted from Xu et al.}
			\end{center}
		\end{column}
		\begin{column}{0.38\textwidth}
			{\large\textbf{What NAS-GS does}}

			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{1.4mm}
				\footnotesize
				\item Learns a 3D Gaussian sonar scene from polar sonar images.
				\item Optimises rendered structure together with acoustic noise appearance.
				\item Produces a differentiable sonar renderer for new viewpoints.
			\end{itemize}

			\vspace{2mm}
			{\large\textbf{How I use it}}

			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{1.4mm}
				\footnotesize
				\item Freeze the trained map during replay.
				\item Render candidate Pose3 hypotheses online.
				\item Insert a factor only when live sonar and render agree.
			\end{itemize}
		\end{column}
	\end{columns}

\end{frame}

			\node[box] (factor) at (11.25,0) {Pose3\\factor};
			\draw[arrow] (seq) -- (route);
			\draw[arrow] (route) -- (drift);
			\draw[arrow] (drift) -- (search);
			\draw[arrow] (search) -- (gate);
			\draw[arrow] (gate) -- (factor);
		\end{tikzpicture}
	\end{center}

	\vspace{0.5mm}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			{\small\textbf{Route prior}}
			{\scriptsize
			\[
			s_t=\frac{t}{N_B-1}, \qquad
			i_t=\mathrm{round}\!\left(s_t(N_A-1)\right)
			\]
			\textit{Variables:} \(t\) is the current revisit frame, \(N_A,N_B\) are the map and revisit route lengths, \(s_t\) is normalised progress, and \(i_t\) is the selected Pass A map index.
			\vspace{0.6mm}
			\begin{itemize}
				\setlength{\itemsep}{0.45mm}
				\item This avoids nearest-pose lookup when route direction changes.
				\item It gives a map window, not the final factor pose.
			\end{itemize}
			}
		\end{column}
		\begin{column}{0.48\textwidth}
			{\small\textbf{Drift prior}}
			{\scriptsize
			\[
			T^{cand}_{B,t}=T^{odom}_{B,t}\oplus c_t,\qquad
			c_t\in\mathcal{E}_{drift}(s_t)
			\]
			\textit{Variables:} \(T^{odom}_{B,t}\) is the live drifted Pose3, \(c_t\) is a sampled correction in \(x,y,z,\psi\), and \(\mathcal{E}_{drift}(s_t)\) is the progress-dependent drift envelope.
			\vspace{0.6mm}
			\begin{itemize}
				\setlength{\itemsep}{0.45mm}
				\item It is learned from Pass A baseline-to-GT errors.
				\item It guides search size, not the final accepted pose.
			\end{itemize}
			}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}
	\frametitle{Results Summary}
	\framesubtitle{The method works best as selective map-aided correction}

	\begin{columns}[T]
		\begin{column}{0.47\textwidth}
			\textbf{Evidence}
			\begin{itemize}
				\item Pass A shows strong same-route correction from a trained NAS-GS map.
				\item Pass A revisit confirms the local policy transfers to a newly recorded repeat route.
				\item Offshore BP replay transfers the same backbone to real sonar appearance.
				\item Accepted factors are interpretable because each one has a rendered-image explanation.
			\end{itemize}
		\end{column}
		\begin{column}{0.47\textwidth}
			\textbf{Boundary}
			\begin{itemize}
				\item Reverse-route revisit improves on average, but endpoint stability is not solved.
				\item Wide search needs sequence confirmation before large corrections are trusted.
				\item Offshore accuracy is reference-consistency against Aqua-SLAM, not independent survey truth.
			\end{itemize}
		\end{column}
	\end{columns}

\end{frame}

	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\textbf{Algorithm}
			\begin{itemize}
				\item Strengthen sequence evidence for large-drift recovery.
				\item Improve route-progress endpoint stability for reverse revisits.
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\textbf{Validation}
			\begin{itemize}
				\item Retrain or fine-tune NAS-GS with corrected trajectories.
				\item Repeat offshore tests with an independent reference if available.
				\item Move from replay to real robot timing, calibration, and safety tests.
			\end{itemize}
		\end{column}
	\end{columns}

\begin{frame}
	\frametitle{Conclusion}

	\Large
	NAS-GS can act as a useful sonar map prior for SLAM.

	\vspace{5mm}
	\normalsize
	The implemented system turns render consistency into selective Pose3 registration factors and fuses them with IMU/DVL odometry through iSAM2.

	\vspace{4mm}
	It is strongest for same-route and repeated-inspection correction. Route-progress revisit and offshore replay show the method transfers, but large-drift recovery and independent real-world validation remain the main next steps.
\end{frame}
