\begin{frame}
	\frametitle{Why Underwater Sonar SLAM Is Hard}
	\framesubtitle{Can a renderable sonar map provide online correction factors?}

	\begin{columns}[T]
		\begin{column}{0.47\textwidth}
			\storycard{ICLBlue!6}{ICLBlue}{1. Problem setting}
			{Underwater robots cannot use GNSS. IMU/DVL odometry therefore drifts over time.}

			\vspace{1.8mm}
			\storycard{ICLBlue!6}{ICLBlue}{2. Sensor limitation}
			{Sonar works underwater, but the image evidence is sparse, ambiguous, and viewpoint-dependent.}

			\vspace{1.8mm}
			\storycard{ICLBlue!6}{ICLBlue}{3. Classical SLAM limitation}
			{Geometric matching becomes fragile under sonar noise and repeated acoustic structure.}
		\end{column}
		\begin{column}{0.47\textwidth}
			\storycard{ICLCream}{ICLBlue}{4. Key insight from 3DGS}
			{A learned Gaussian scene can predict the sonar view from a candidate pose.}

			\vspace{1.8mm}
			\storycard{ICLCream}{ICLBlue}{5. Gap}
			{3DGS is rarely used as an online sonar measurement model inside SLAM.}

			\vspace{1.8mm}
			\storycard{ICLCream}{ICLBlue}{6. Project idea}
			{Use render consistency between live sonar and NAS-GS prediction as a Pose3 SLAM constraint.}
		\end{column}
	\end{columns}

\end{frame}

\begin{frame}
	\frametitle{NAS-GS: Noise-Aware Sonar Gaussian Splatting}
	\framesubtitle{A renderable soanr measurement model}

	\begin{columns}[T]
		\begin{column}{0.58\textwidth}
			\vspace{-1mm}
			\begin{center}
				\includegraphics[width=\linewidth]{nasgs_pipeline_overview.png}

				\vspace{0.5mm}
				{\scriptsize NAS-GS training pipeline, adapted from Xu et al.}
			\end{center}
		\end{column}
		\begin{column}{0.38\textwidth}
			{\large\textbf{What NAS-GS does}}

			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{1.4mm}
				\footnotesize
				\item Learns a 3D Gaussian sonar scene from polar sonar images.
				\item Optimises rendered structure together with acoustic noise appearance.
				\item Produces a differentiable sonar renderer for new viewpoints.
			\end{itemize}

			\vspace{2mm}
			{\large\textbf{How I use it}}

			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{1.4mm}
				\footnotesize
				\item Freeze the trained map during replay.
				\item Render candidate Pose3 hypotheses online.
				\item Insert a factor only when live sonar and render agree.
			\end{itemize}
		\end{column}
	\end{columns}

\end{frame}

\begin{frame}
	\frametitle{Algorithm}
	\framesubtitle{Candidate poses are searched first; the graph is changed only after gating}

	\begin{columns}[T]
		\begin{column}{0.47\textwidth}
			{\large\textbf{Candidate poses}}
			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{1.6mm}
				\footnotesize
				\item Start from the live IMU/DVL Pose3 estimate.
				\item Sample small corrections in \(x,y,z\) and yaw.
				\item Local mode searches a compact recoverable basin.
				\item Revisit mode uses a wider sequence proposal guided by route progress and drift prior.
				\item A candidate is always a possible robot pose. The NAS-GS map is only used to render evidence.
			\end{itemize}
		\end{column}
		\begin{column}{0.47\textwidth}
			{\large\textbf{Score and gate}}
			\vspace{1mm}
			\begin{itemize}
				\setlength{\itemsep}{1.6mm}
				\footnotesize
				\item Each candidate is rendered by the fixed NAS-GS map.
				\item Image score combines NCC-style similarity with error after intensity normalisation.
				\item Structure score checks that bright sonar returns and rendered returns overlap.
				\item Gates reject low score, weak structure, excessive pose jump, and unstable sequence support.
				\item The best passing candidate provides the Pose3 factor pose and covariance.
			\end{itemize}
		\end{column}
	\end{columns}

	\vspace{4mm}
	\begin{center}
		\footnotesize
		The key design choice is conservative evidence: a render match can correct drift, but a failed match does not damage the pose graph.
	\end{center}
\end{frame}

\begin{frame}
	\frametitle{What The Results Show}

	\begin{columns}[T]
		\begin{column}{0.47\textwidth}
			\textbf{Strengths}
			\begin{itemize}
				\item Local same-route correction works.
				\item The pipeline runs as a live replay.
				\item Offshore validation shows real-scene relevance.
				\item Render evidence gives interpretable factor placement.
			\end{itemize}
		\end{column}
		\begin{column}{0.47\textwidth}
			\textbf{Limits}
			\begin{itemize}
				\item Sonar ambiguity can accept visually plausible but wrong candidates.
				\item z correction is sensitive to rendering and trajectory priors.
				\item Hard revisit remains limited without stronger sequence geometry.
				\item Render-server geometry must match the sonar dataset.
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

%------------------------------------------------

\begin{frame}
	\frametitle{Conclusion}

	\Large
	This project demonstrates an online-render SLAM correction system.

	\vspace{5mm}
	\normalsize
	The main result is that a NAS-GS sonar map can provide useful Pose3 registration factors when the live sonar appearance, render geometry, and search basin are aligned.

	\vspace{4mm}
	The main limitation is revisit robustness. The next step is a stronger sequence-level solver that estimates smooth candidate paths rather than accepting isolated frame evidence.
\end{frame}
